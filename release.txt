# Releasing a new version

# sanity check
  make rcheck
  
# versioning
# edit version in setup.cfg
  git commit -m 'update version' setup.cfg
  git push

# update the version in setup.cfg and in docs/conf.py
  git commit -m 'update version' setup.cfg docs/conf.py

# update CHANGELOG.rst  `git shortlog v0.6.0..HEAD`
  git commit -m 'update recent changes' CHANGELOG.rst
  git push

# create release tag on github
  git tag v1.0.1
  git push origin v1.0.1

# upload source to pypi
  rm -rf dist/*
  python setup.py sdist
  twine upload dist/*

# test the install by removing local developer version
# installing, then uninstalling, then restoring developer version
  pip install pypolar
  pip uninstall pypolar
